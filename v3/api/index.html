# 生命周期

仅在setup函数内部有效

import {onMounted,onUpdated,onUnmounted} from 'vue'

const myComponent={
setup(){
    // 组件挂载触发
    onMounted(()=>{
        console.log('第一步: 组件挂载')
    })
    // 组件有更新
    onUpdated(()=>{
        console.log('第二步：组件有更新')
    })
    // 卸载组件
    onUnmounted(()=>{
        console.log('第三步：卸载组件')
    })
}

}
